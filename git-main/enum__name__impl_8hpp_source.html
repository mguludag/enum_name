171template <typename Enum, int Min, int Max>
172MGUTILITY_CNSTXPR inline auto to_enum_impl(mgutility::string_view str) noexcept
173 -> mgutility::optional<Enum> {
174 MGUTILITY_CNSTXPR auto arr = get_enum_array<Enum, Min, Max>();
175 const auto index{std::find(arr.begin() + 1, arr.end(), str)};
176 return index == arr.end() ? mgutility::nullopt
177 : mgutility::optional<Enum>{static_cast<Enum>(
178 std::distance(arr.begin(), index) + Min - 1)};
179}

190template <typename Enum, int Min, int Max>
191MGUTILITY_CNSTXPR auto to_enum_bitmask_impl(mgutility::string_view str) noexcept
192 -> mgutility::optional<Enum> {
193
194 // Check if the string contains a '|' character
195 if (str.find('|') == mgutility::string_view::npos) {
196 return to_enum_impl<Enum, Min, Max>(str);
197 }
198
199 mgutility::optional<Enum> result{mgutility::nullopt};
200 std::size_t index = 0;
201
202 for (std::size_t i = 0; i < str.size(); ++i) {
203 if (str[i] == '|') {
204 auto name = str.substr(index, i - index);
205 auto maybe_enum = to_enum_impl<Enum, Min, Max>(name);
206
207 if (!name.empty() && maybe_enum) {
208 result.emplace(result ? static_cast<Enum>(*result | *maybe_enum)
209 : *maybe_enum);
210 }
211
212 index = i + 1;
213 }
214 }
215
216 auto maybe_enum = to_enum_impl<Enum, Min, Max>(str.substr(index));
217 if (result && maybe_enum) {
218 result.emplace(static_cast<Enum>(*result | *maybe_enum));
219 } else {
220 result.reset();
221 }
222
223 return result;
224}

235template <typename Enum, int Min, int Max,
236 detail::enable_if_t<!detail::has_bit_or<Enum>::value, bool> = true>
237MGUTILITY_CNSTXPR auto enum_name_impl(Enum e) noexcept
238 -> mgutility::string_view {
239 MGUTILITY_CNSTXPR auto arr = get_enum_array<Enum, Min, Max>();
240 const auto index{(Min < 0 ? Min * -1 : Min) + static_cast<int>(e) + 1};
241 return arr[(index < Min || index > arr.size() - 1) ? 0 : index];

254template <typename Enum, int Min, int Max,
255 detail::enable_if_t<detail::has_bit_or<Enum>::value, bool> = true>
256MGUTILITY_CNSTXPR_CLANG_WA inline auto enum_name_impl(Enum e) noexcept
257 -> detail::string_or_view_t<Enum> {
258
259 // Get the array of enum names
260 MGUTILITY_CNSTXPR auto arr = get_enum_array<Enum, Min, Max>();
261
262 // Calculate the index in the array
263 const auto index = (Min < 0 ? -Min : Min) + static_cast<int>(e) + 1;
264 const auto name =
265 arr[(index < Min || index >= static_cast<int>(arr.size())) ? 0 : index];
266
267 // Lambda to check if a character is a digit
268 const auto is_digit = [](char c) { return c >= '0' && c <= '9'; };
269
270 // Return the name if it's valid
271 if (!name.empty() && !is_digit(name[0])) {
272 return std::string{name};
273 }
274
275 // Construct bitmasked name
276 std::string bitmasked_name;
277 for (auto i = Min; i < Max; ++i) {
278 const auto idx = (Min < 0 ? -Min : Min) + i + 1;
279 if (idx >= 0 && idx < static_cast<int>(arr.size()) && !arr[idx].empty() &&
280 !is_digit(arr[idx][0]) &&
281 (e & static_cast<Enum>(i)) == static_cast<Enum>(i)) {
282 bitmasked_name.append(arr[idx]).append("|");
283 }
284 }
285
286 // Remove the trailing '|' if present
287 if (!bitmasked_name.empty()) {
288 bitmasked_name.pop_back();
289 }
290
291 return bitmasked_name.find('|') != std::string::npos ? bitmasked_name
292 : std::string{""};
293}
