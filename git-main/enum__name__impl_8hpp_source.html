135 MGUTILITY_CNSTXPR static mgutility::string_view
136 parse(mgutility::string_view str) noexcept {
137#if defined(__clang__) || defined(__GNUC__)
138#if defined(__clang__)
139 auto end = str.rfind(']');
140#elif defined(__GNUC__) && !defined(__clang__)
141 auto end = str.rfind(';');
142#endif
143 // Typical form:
144 // "Enum = MyEnum::Value]"
145
146 auto pos = str.rfind('=', end);
147 if (pos == mgutility::string_view::npos) {
148 return {};
149 }
150 pos += 2; // skip "::"
151
152 auto result = str.substr(pos, end - pos);
153
154#elif defined(_MSC_VER)
155 // MSVC: different format
156 auto pos = str.rfind(',');
157 if (pos == mgutility::string_view::npos)
158 return {};
159
160 ++pos;
161
162 auto end = str.rfind('>');
163 auto result = str.substr(pos, end - pos);
164
165#else
166 return {};
167#endif
168
169 if (result.empty()) {
170 return {};
171 }
172
173 // invalid cases look like "(Enum)123"
174 if (result[0] == '(') {
175 return {};
176 }
177
178 return result.substr(result.rfind(':') + 1);
179 }

190 MGUTILITY_CNSTXPR static mgutility::string_view name() noexcept {
191 return parse(raw_name<Enum, e>());

209template <typename Enum, Enum... Is>

211#if MGUTILITY_CPLUSPLUS >= 201402L
212 // C++17+: fully constexpr
213 // NOLINTNEXTLINE [readability-redundant-inline-specifier]
214 static inline constexpr std::array<mgutility::string_view, sizeof...(Is) + 1>
215 value() {
216 std::array<mgutility::string_view, sizeof...(Is) + 1> arr{
217 "", enum_type::template name<Enum, Is>()...};
218
219 constexpr auto map = mgutility::custom_enum<Enum>::map;
220
221 for (const auto &pair : map) {
222 const int raw = static_cast<int>(pair.first);
223 const auto idx =
224 static_cast<size_t>(raw - mgutility::enum_range<Enum>::min + 1);
225
226 if (idx >= 1 && idx < arr.size()) {
227 arr[idx] = pair.second;
228 }
229 }
230
231 return arr;
232 }
233#else
234 // C++11: lazy runtime array
235 static const std::array<mgutility::string_view, sizeof...(Is) + 1> &value() {
236 static const std::array<mgutility::string_view, sizeof...(Is) + 1> arr =
237 [] {
238 std::array<mgutility::string_view, sizeof...(Is) + 1> tmp{

309template <typename Enum, int Min, int Max>
310MGUTILITY_CNSTXPR inline auto to_enum_impl(mgutility::string_view str) noexcept
311 -> mgutility::optional<Enum> {
312 MGUTILITY_CNSTXPR_CLANG_WA auto arr = get_enum_array<Enum, Min, Max>();
313
314 const auto index{detail::find(arr, str)};
315 return index == 0
316 ? mgutility::nullopt
317 : mgutility::optional<Enum>{static_cast<Enum>(index + Min - 1)};
318}

329template <typename Enum, int Min, int Max>
330MGUTILITY_CNSTXPR auto to_enum_bitmask_impl(mgutility::string_view str) noexcept
331 -> mgutility::optional<Enum> {
332
333 // Check if the string contains a '|' character
334 if (str.find('|') == mgutility::string_view::npos) {
335 return to_enum_impl<Enum, Min, Max>(str);
336 }
337
338 mgutility::optional<Enum> result{mgutility::nullopt};
339 std::size_t index = 0;
340
341 for (std::size_t i = 0; i < str.size(); ++i) {
342 if (str[i] == '|') {
343 auto name = str.substr(index, i - index);
344 auto maybe_enum = to_enum_impl<Enum, Min, Max>(name);
345
346 if (!name.empty() && maybe_enum) {
347 result.emplace(result ? static_cast<Enum>(*result | *maybe_enum)
348 : *maybe_enum);
349 }
350
351 index = i + 1;
352 }
353 }
354
355 auto maybe_enum = to_enum_impl<Enum, Min, Max>(str.substr(index));
356 if (result && maybe_enum) {
357 result.emplace(static_cast<Enum>(*result | *maybe_enum));
358 } else {
359 result.reset();
360 }
361
362 return result;
363}

374template <typename Enum, int Min, int Max,
375 detail::enable_if_t<!detail::has_bit_or<Enum>::value, bool> = true>
376MGUTILITY_CNSTXPR auto enum_name_impl(Enum enumValue) noexcept
377 -> mgutility::string_view {
378 MGUTILITY_CNSTXPR_CLANG_WA auto arr = get_enum_array<Enum, Min, Max>();
379 const auto index{(Min < 0 ? -Min : Min) + static_cast<int>(enumValue) + 1};
380 return arr[static_cast<size_t>(
381 (index < Min || index > static_cast<int>(arr.size()) - 1) ? 0 : index)];
382}

395template <typename Enum, int Min, int Max,
396 detail::enable_if_t<detail::has_bit_or<Enum>::value, bool> = true>
397MGUTILITY_CNSTXPR_CLANG_WA auto enum_name_impl(Enum enumValue) noexcept
398 -> mgutility::fixed_string<enum_name_buffer<Enum>::size> {
399
400 // Get the array of enum names
401 MGUTILITY_CNSTXPR_CLANG_WA auto arr = get_enum_array<Enum, Min, Max>();
402
403 // Calculate the index in the array
404 const auto index = (Min < 0 ? -Min : Min) + static_cast<int>(enumValue) + 1;
405 const auto name =
406 arr[(index < Min || index >= static_cast<int>(arr.size())) ? 0 : index];
407
408 // Return the name if it's valid
409 if (!name.empty() && !is_digit(name[0])) {
410 return mgutility::fixed_string<enum_name_buffer<Enum>::size>{}.append(name);
411 }
412
413 // Construct bitmasked name
414 mgutility::fixed_string<enum_name_buffer<Enum>::size> bitmasked_name;
415 for (auto i = Min; i < Max; ++i) {
416 const auto idx = (Min < 0 ? -Min : Min) + i + 1;
417 if (idx >= 0 && idx < static_cast<int>(arr.size()) && !arr[idx].empty() &&
418 !detail::is_digit(arr[idx][0]) &&
419 (enumValue & static_cast<Enum>(i)) == static_cast<Enum>(i)) {
420 bitmasked_name.append(arr[idx]).append("|");
421 }
422 }
423
424 // Remove the trailing '|' if present
425 if (!bitmasked_name.empty()) {
426 bitmasked_name.pop_back();
427 }
428
429 return bitmasked_name;
430}
