76MGUTILITY_CNSTXPR int strncmp_constexpr(const char *lhs, const char *rhs,
77 size_t count) noexcept {
78 for (size_t i = 0; i < count; ++i) {
79 // NOLINTNEXTLINE [cppcoreguidelines-pro-bounds-pointer-arithmetic]
80 if (lhs[i] != rhs[i] || lhs[i] == '\0' || rhs[i] == '\0') {
81 // NOLINTNEXTLINE [cppcoreguidelines-pro-bounds-pointer-arithmetic]
82 return static_cast<unsigned char>(lhs[i]) -
83 // NOLINTNEXTLINE [cppcoreguidelines-pro-bounds-pointer-arithmetic]
84 static_cast<unsigned char>(rhs[i]);
85 }
86 }
87 return 0;
88}
