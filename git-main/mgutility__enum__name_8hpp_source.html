121// NOLINTNEXTLINE [readability-identifier-length]
122constexpr auto strlen_constexpr(const char *str, size_t sz = 0) noexcept
123 -> size_t {
124 // NOLINTNEXTLINE [cppcoreguidelines-pro-bounds-pointer-arithmetic]
125 return str[sz] == '\0' ? sz : strlen_constexpr(str, ++sz);
126}
127
134constexpr auto is_digit(char character) noexcept -> bool {
135 return character >= '0' && character <= '9';
136}
137
147MGUTILITY_CNSTXPR int strncmp_constexpr(const char *lhs, const char *rhs,
148 size_t count) noexcept {
149 for (size_t i = 0; i < count; ++i) {
150 // NOLINTNEXTLINE [cppcoreguidelines-pro-bounds-pointer-arithmetic]
151 if (lhs[i] != rhs[i] || lhs[i] == '\0' || rhs[i] == '\0') {
152 // NOLINTNEXTLINE [cppcoreguidelines-pro-bounds-pointer-arithmetic]
153 return static_cast<unsigned char>(lhs[i]) -
154 // NOLINTNEXTLINE [cppcoreguidelines-pro-bounds-pointer-arithmetic]
155 static_cast<unsigned char>(rhs[i]);
156 }
157 }
158 return 0;
159}
160
161template <typename Range, typename Pred>
162MGUTILITY_CNSTXPR auto find(const Range &rng, const Pred &pred) -> size_t {
163 for (size_t i = 1; i < rng.size(); ++i) {
164 if (pred == rng[i]) {
165 return i;
166 }
167 }
168 return 0;
169};

451template <size_t N = 0> class fixed_string {
452public:
453 template <size_t M>
454 // NOLINTNEXTLINE [cppcoreguidelines-avoid-c-arrays]
455 MGUTILITY_CNSTXPR static auto make(const char (&str)[M]) -> fixed_string<M> {
456 return fixed_string<M>{str};
457 }
458
459 MGUTILITY_CNSTXPR fixed_string() = default;
460
461 // Constructor to initialize from a string literal
462 // NOLINTNEXTLINE [cppcoreguidelines-avoid-c-arrays]
463 MGUTILITY_CNSTXPR explicit fixed_string(const char (&str)[N]) {
464 for (size_t i = 0; i < N - 1; ++i) {
465 data[i] = str[i];
466 }
467 cursor = N - 1;
468 data[cursor] = '\0';
469 }
470
471 // Concatenation operator
472 template <size_t M>
473 MGUTILITY_CNSTXPR auto operator+(const fixed_string<M> &other) const
474 -> fixed_string<N + M - 1> {
475 fixed_string<N + M - 1> result{};
476 size_t idx = 0;
477 for (; idx < N - 1; ++idx) {
478 // NOLINTNEXTLINE [cppcoreguidelines-pro-bounds-constant-array-index]
479 result.data[idx] = data[idx];
480 }
481 for (size_t j = 0; j < M; ++j) {
482 // NOLINTNEXTLINE [cppcoreguidelines-pro-bounds-constant-array-index]
483 result.data[idx + j] = other.data[j];
484 }
485 result.cursor = N + M - 2;
486 // NOLINTNEXTLINE [cppcoreguidelines-pro-bounds-constant-array-index]
487 result.data[result.cursor] = '\0';
488 return result;
489 }
490
491 // Concatenation operator
492 template <size_t M>
493 // NOLINTNEXTLINE [cppcoreguidelines-avoid-c-arrays]
494 MGUTILITY_CNSTXPR auto operator+(const char (&str)[M]) const
495 -> fixed_string<N + M - 1> {
496 return *this + fixed_string<M>{str};
497 }
498
499 template <size_t M>
500 // NOLINTNEXTLINE [cppcoreguidelines-avoid-c-arrays]
501 MGUTILITY_CNSTXPR auto append(const char (&str)[M]) -> fixed_string<N> & {
502 static_assert(N > M,
503 "Capacity needs to be greater than string to be appended!");
504 for (size_t i = 0; i < M - 1; ++i) {
505 // NOLINTNEXTLINE [cppcoreguidelines-pro-bounds-constant-array-index]
506 data[cursor++] = str[i];
507 }
508 // NOLINTNEXTLINE [cppcoreguidelines-pro-bounds-constant-array-index]
509 data[cursor] = '\0';
510 return *this;
511 }
512
513 MGUTILITY_CNSTXPR auto append(string_view str) -> fixed_string<N> & {
514 for (const char chr : str) {
515 // NOLINTNEXTLINE [cppcoreguidelines-pro-bounds-constant-array-index]
516 data[cursor++] = chr;
517 }
518 // NOLINTNEXTLINE [cppcoreguidelines-pro-bounds-constant-array-index]
519 data[cursor] = '\0';
520 return *this;
521 }
522
523 MGUTILITY_CNSTXPR auto pop_back() -> void {
524 if (cursor > 0) {
525 // NOLINTNEXTLINE [cppcoreguidelines-pro-bounds-constant-array-index]
526 data[--cursor] = '\0';
527 }
528 }
529
530 MGUTILITY_CNSTXPR auto size() const -> size_t { return cursor; }
531
532 // NOLINTNEXTLINE [readability-identifier-length]
533 constexpr size_t find(char c, size_t pos = 0) const noexcept {
534 // NOLINTNEXTLINE [readability-avoid-nested-conditional-operator]
535 return c == data[pos] ? pos : (pos < cursor ? find(c, ++pos) : npos);
536 }
537
538 // Conversion to std::string_view for easy printing
539 // NOLINTNEXTLINE[google-explicit-constructor]
540 MGUTILITY_CNSTXPR operator string_view() const {
541 return string_view(data, cursor);
542 }
543
544 MGUTILITY_CNSTXPR auto view() const -> string_view {
545 return string_view(data, cursor);
546 }
547
548 constexpr bool empty() const noexcept { return cursor == 0; }
549
550 constexpr const char &operator[](size_t index) const noexcept {
551 // NOLINTNEXTLINE [cppcoreguidelines-pro-bounds-constant-array-index]
552 return data[index];
553 }
554
555 MGUTILITY_CNSTXPR inline bool operator==(const char *rhs) const {
556 return view() == rhs;
557 }
558
559 // NOLINTNEXTLINE [readability-identifier-length]
560 friend std::ostream &operator<<(std::ostream &os,
561 const fixed_string<N> &str) {
562 for (size_t i = 0; i < str.cursor; ++i) {
563 // NOLINTNEXTLINE [cppcoreguidelines-pro-bounds-constant-array-index]
564 os << str.data[i];
565 }
566 return os;
567 }
568
569 static constexpr auto npos = -1;
570
571private:
572 // NOLINTNEXTLINE [cppcoreguidelines-avoid-c-arrays]
573 char data[N]{'\0'};
574 size_t cursor{};
575};

1546MGUTILITY_CNSTXPR inline auto to_enum_impl(mgutility::string_view str) noexcept
1547 -> mgutility::optional<Enum> {
1548 MGUTILITY_CNSTXPR_CLANG_WA auto arr = get_enum_array<Enum, Min, Max>();
1549

1565template <typename Enum, int Min, int Max>
1566MGUTILITY_CNSTXPR auto to_enum_bitmask_impl(mgutility::string_view str) noexcept
1567 -> mgutility::optional<Enum> {
1568
1569 // Check if the string contains a '|' character
1570 if (str.find('|') == mgutility::string_view::npos) {
1571 return to_enum_impl<Enum, Min, Max>(str);
1572 }
1573
1574 mgutility::optional<Enum> result{mgutility::nullopt};
1575 std::size_t index = 0;
1576
1577 for (std::size_t i = 0; i < str.size(); ++i) {
1578 if (str[i] == '|') {
1579 auto name = str.substr(index, i - index);
1580 auto maybe_enum = to_enum_impl<Enum, Min, Max>(name);
1581
1582 if (!name.empty() && maybe_enum) {
1583 result.emplace(result ? static_cast<Enum>(*result | *maybe_enum)
1584 : *maybe_enum);
1585 }
1586
1587 index = i + 1;
1588 }
1589 }
1590
1591 auto maybe_enum = to_enum_impl<Enum, Min, Max>(str.substr(index));
1592 if (result && maybe_enum) {
1593 result.emplace(static_cast<Enum>(*result | *maybe_enum));
1594 } else {
1595 result.reset();
1596 }
1597
1598 return result;
1599}

1630template <typename Enum, int Min, int Max,
1631 detail::enable_if_t<detail::has_bit_or<Enum>::value, bool> = true>
1632MGUTILITY_CNSTXPR_CLANG_WA auto enum_name_impl(Enum enumValue) noexcept
1633 -> fixed_string<enum_name_buffer<Enum>::size> {
1634
1635 // Get the array of enum names
1636 MGUTILITY_CNSTXPR_CLANG_WA auto arr = get_enum_array<Enum, Min, Max>();
1637
1638 // Calculate the index in the array
1639 const auto index = (Min < 0 ? -Min : Min) + static_cast<int>(enumValue) + 1;
1640 const auto name =
1641 arr[(index < Min || index >= static_cast<int>(arr.size())) ? 0 : index];
1642
1643 // Return the name if it's valid
1644 if (!name.empty() && !is_digit(name[0])) {
1645 return fixed_string<enum_name_buffer<Enum>::size>{}.append(name);
1646 }
1647
1648 // Construct bitmasked name
1649 fixed_string<enum_name_buffer<Enum>::size> bitmasked_name;
1650 for (auto i = Min; i < Max; ++i) {
1651 const auto idx = (Min < 0 ? -Min : Min) + i + 1;
1652 if (idx >= 0 && idx < static_cast<int>(arr.size()) && !arr[idx].empty() &&
1653 !detail::is_digit(arr[idx][0]) &&
1654 (enumValue & static_cast<Enum>(i)) == static_cast<Enum>(i)) {
1655 bitmasked_name.append(arr[idx]).append("|");
1656 }
1657 }
1658
1659 // Remove the trailing '|' if present
1660 if (!bitmasked_name.empty()) {
1661 bitmasked_name.pop_back();
1662 }
1663
