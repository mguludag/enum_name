67 MGUTILITY_CNSTXPR int strncmp_constexpr(const char *lhs, const char *rhs, size_t count) noexcept {
68 for (size_t i = 0; i < count; ++i) {
69 if (lhs[i] != rhs[i] || lhs[i] == '\0' || rhs[i] == '\0') {
70 return static_cast<unsigned char>(lhs[i]) - static_cast<unsigned char>(rhs[i]);
71 }
72 }
73 return 0;
74}

346template <size_t N = 0> class fixed_string {
347public:
348 template <size_t M>
349 MGUTILITY_CNSTXPR static auto make(const char (&str)[M]) -> fixed_string<M> {
350 return fixed_string<M>{str};
351 }
352
353 MGUTILITY_CNSTXPR fixed_string() = default;
354
355 // Constructor to initialize from a string literal
356 MGUTILITY_CNSTXPR explicit fixed_string(const char (&str)[N]) {
357 for (size_t i = 0; i < N - 1; ++i) {
358 data[i] = str[i];
359 }
360 cursor = N - 1;
361 data[cursor] = '\0';
362 }
363
364 // Concatenation operator
365 template <size_t M>
366 MGUTILITY_CNSTXPR auto operator+(const fixed_string<M> &other) const
367 -> fixed_string<N + M - 1> {
368 fixed_string<N + M - 1> result{};
369 size_t idx = 0;
370 for (; idx < N - 1; ++idx) {
371 result.data[idx] = data[idx];
372 }
373 for (size_t j = 0; j < M; ++j) {
374 result.data[idx + j] = other.data[j];
375 }
376 result.cursor = N + M - 2;
377 result.data[result.cursor] = '\0';
378 return result;
379 }
380
381 // Concatenation operator
382 template <size_t M>
383 MGUTILITY_CNSTXPR auto operator+(const char (&str)[M]) const
384 -> fixed_string<N + M - 1> {
385 return *this + fixed_string<M>{str};
386 }
387
388 template <size_t M>
389 MGUTILITY_CNSTXPR auto append(const char (&str)[M]) -> fixed_string<N> & {
390 static_assert(N > M,
391 "Capacity needs to be greater than string to be appended!");
392 for (size_t i = 0; i < M - 1; ++i) {
393 data[cursor++] = str[i];
394 }
395 data[cursor] = '\0';
396 return *this;
397 }
398
399 MGUTILITY_CNSTXPR auto append(string_view str) -> fixed_string<N> & {
400 for (char chr : str) {
401 data[cursor++] = chr;
402 }
403 data[cursor] = '\0';
404 return *this;
405 }
406
407 MGUTILITY_CNSTXPR auto pop_back() -> void {
408 if (cursor > 0) {
409 data[--cursor] = '\0';
410 }
411 }
412
413 MGUTILITY_CNSTXPR auto size() const -> size_t { return cursor; }
414
415 constexpr size_t find(char c, size_t pos = 0) const noexcept {
416 return c == data[pos] ? pos : (pos < cursor ? find(c, ++pos) : npos);
417 }
418
419 // Conversion to std::string_view for easy printing
420 MGUTILITY_CNSTXPR explicit operator string_view() const {
421 return string_view(data, cursor);
422 }
423
424 constexpr bool empty() const noexcept { return cursor == 0; }
425
426 constexpr const char &operator[](size_t index) const noexcept {
427 return data[index];
428 }
429
430 MGUTILITY_CNSTXPR inline bool operator==(const char *rhs) const {
431 return string_view(*this) == rhs;
432 }
433
434 friend std::ostream &operator<<(std::ostream &os,
435 const fixed_string<N> &str) {
436 for (size_t i = 0; i < str.cursor; ++i) {
437 os << str.data[i];
438 }
439 return os;
440 }
441
442 static constexpr auto npos = -1;
443
444private:
445 char data[N]{'\0'};
446 size_t cursor{};
447};
