113 template <typename Enum, Enum e>
114 MGUTILITY_CNSTXPR static mgutility::string_view raw_name() noexcept {
115 return mgutility::string_view(__PRETTY_FUNCTION__,
116 sizeof(__PRETTY_FUNCTION__) - 1);

125 MGUTILITY_CNSTXPR static mgutility::string_view
126 parse(mgutility::string_view str) noexcept {
127#if defined(__clang__) || defined(__GNUC__)
128#if defined(__clang__)
129 auto end = str.rfind(']');
130#elif defined(__GNUC__) && !defined(__clang__)
131 auto end = str.rfind(';');
132#endif
133 // Typical form:
134 // "Enum = MyEnum::Value]"
135
136 auto pos = str.rfind('=', end);
137 if (pos == mgutility::string_view::npos) {
138 return {};
139 }
140 pos += 2; // skip "::"
141
142 auto result = str.substr(pos, end - pos);
143
144#elif defined(_MSC_VER)
145 // MSVC: different format
146 auto pos = str.rfind(',');
147 if (pos == mgutility::string_view::npos)
148 return {};
149
150 ++pos;
151
152 auto end = str.rfind('>');
153 auto result = str.substr(pos, end - pos);
154
155#else
156 return {};
157#endif
158
159 if (result.empty()) {
160 return {};
161 }
162
163 // invalid cases look like "(Enum)123"
164 if (result[0] == '(') {
165 return {};
166 }
167
168 return result.substr(result.rfind(':') + 1);
169 }
170
171public:
179 template <typename Enum, Enum e>
180 MGUTILITY_CNSTXPR static mgutility::string_view name() noexcept {
181 // 1. Custom override first
182 auto custom = lookup_custom(e);
183 if (!custom.empty()) {
184 return custom;
185 }
186
187 // 2. Extract + parse
188 return parse(raw_name<Enum, e>());
189 }

276template <typename Enum, int Min, int Max>
277MGUTILITY_CNSTXPR inline auto to_enum_impl(mgutility::string_view str) noexcept
278 -> mgutility::optional<Enum> {
279 MGUTILITY_CNSTXPR_CLANG_WA auto arr = get_enum_array<Enum, Min, Max>();
280
281 const auto index{detail::find(arr, str)};
282 return index == 0
283 ? mgutility::nullopt
284 : mgutility::optional<Enum>{static_cast<Enum>(index + Min - 1)};
285}

296template <typename Enum, int Min, int Max>
297MGUTILITY_CNSTXPR auto to_enum_bitmask_impl(mgutility::string_view str) noexcept
298 -> mgutility::optional<Enum> {
299
300 // Check if the string contains a '|' character
301 if (str.find('|') == mgutility::string_view::npos) {
302 return to_enum_impl<Enum, Min, Max>(str);
303 }
304
305 mgutility::optional<Enum> result{mgutility::nullopt};
306 std::size_t index = 0;
307
308 for (std::size_t i = 0; i < str.size(); ++i) {
309 if (str[i] == '|') {
310 auto name = str.substr(index, i - index);
311 auto maybe_enum = to_enum_impl<Enum, Min, Max>(name);
312
313 if (!name.empty() && maybe_enum) {
314 result.emplace(result ? static_cast<Enum>(*result | *maybe_enum)
315 : *maybe_enum);
316 }
317
318 index = i + 1;
319 }
320 }
321
322 auto maybe_enum = to_enum_impl<Enum, Min, Max>(str.substr(index));
323 if (result && maybe_enum) {
324 result.emplace(static_cast<Enum>(*result | *maybe_enum));
325 } else {
326 result.reset();
327 }
328
329 return result;
330}

341template <typename Enum, int Min, int Max,
342 detail::enable_if_t<!detail::has_bit_or<Enum>::value, bool> = true>
343MGUTILITY_CNSTXPR auto enum_name_impl(Enum enumValue) noexcept
344 -> mgutility::string_view {
345 MGUTILITY_CNSTXPR_CLANG_WA auto arr = get_enum_array<Enum, Min, Max>();
346 const auto index{(Min < 0 ? -Min : Min) + static_cast<int>(enumValue) + 1};
347 return arr[static_cast<size_t>(
348 (index < Min || index > static_cast<int>(arr.size()) - 1) ? 0 : index)];
349}

362template <typename Enum, int Min, int Max,
363 detail::enable_if_t<detail::has_bit_or<Enum>::value, bool> = true>
364MGUTILITY_CNSTXPR_CLANG_WA auto enum_name_impl(Enum enumValue) noexcept
365 -> mgutility::fixed_string<enum_name_buffer<Enum>::size> {
366
367 // Get the array of enum names
368 MGUTILITY_CNSTXPR_CLANG_WA auto arr = get_enum_array<Enum, Min, Max>();
369
370 // Calculate the index in the array
371 const auto index = (Min < 0 ? -Min : Min) + static_cast<int>(enumValue) + 1;
372 const auto name =
373 arr[(index < Min || index >= static_cast<int>(arr.size())) ? 0 : index];
374
375 // Return the name if it's valid
376 if (!name.empty() && !is_digit(name[0])) {
377 return mgutility::fixed_string<enum_name_buffer<Enum>::size>{}.append(name);
378 }
379
380 // Construct bitmasked name
381 mgutility::fixed_string<enum_name_buffer<Enum>::size> bitmasked_name;
382 for (auto i = Min; i < Max; ++i) {
383 const auto idx = (Min < 0 ? -Min : Min) + i + 1;
384 if (idx >= 0 && idx < static_cast<int>(arr.size()) && !arr[idx].empty() &&
385 !detail::is_digit(arr[idx][0]) &&
386 (enumValue & static_cast<Enum>(i)) == static_cast<Enum>(i)) {
387 bitmasked_name.append(arr[idx]).append("|");
388 }
389 }
390
391 // Remove the trailing '|' if present
392 if (!bitmasked_name.empty()) {
393 bitmasked_name.pop_back();
394 }
395
396 return bitmasked_name;
397}
