144 template <typename Enum, Enum e>
145 MGUTILITY_CNSTXPR static string_type<Enum> parse() noexcept {
146 auto str = raw_name<Enum, e>();
147#if defined(__clang__) || defined(__GNUC__)
148#if defined(__clang__)
149 auto end = str.rfind(']');
150#elif defined(__GNUC__) && !defined(__clang__)
151 auto end = str.rfind(';');
152#endif
153 // Typical form:
154 // "Enum = MyEnum::Value]"
155
156 auto pos = str.rfind('=', end);
157 if (pos == mgutility::string_view::npos) {
158 return {};
159 }
160 pos += 2; // skip "::"
161
162 auto result = str.substr(pos, end - pos);
163
164#elif defined(_MSC_VER)
165 // MSVC: different format
166 auto pos = str.rfind(',');
167 if (pos == mgutility::string_view::npos)
168 return {};
169
170 ++pos;
171
172 auto end = str.rfind('>');
173 auto result = str.substr(pos, end - pos);
174
175#else
176 return {};
177#endif
178
179 if (result.empty()) {
180 return {};
181 }
182
183 // invalid cases look like "(Enum)123"
184 if (result[0] == '(') {
185 return {};
186 }
187
188 return string_type<Enum>(result.substr(result.rfind(':') + 1));
189 }

199 template <typename Enum, Enum e>
200 MGUTILITY_CNSTXPR static string_type<Enum> name() noexcept {
201 return parse<Enum, e>();
202 }

221#if MGUTILITY_CPLUSPLUS >= 201402L

223 // NOLINTNEXTLINE [readability-redundant-inline-specifier]
224 static inline constexpr std::array<enum_type::string_type<Enum>,
225 sizeof...(Is) + 1>
226 value() {
227 std::array<enum_type::string_type<Enum>, sizeof...(Is) + 1> arr{
228 "", enum_type::template name<Enum, Is>()...};
229
230 constexpr auto map = mgutility::custom_enum<Enum>::map;
231
232 for (const auto &pair : map) {
233 const int raw = static_cast<int>(pair.first);
234 const auto idx = static_cast<size_t>(raw - Min + 1);
235
236 if (idx >= 1 && idx < arr.size()) {
237 arr[idx] = enum_type::string_type<Enum>(pair.second);
238 }
239 }
240
241 return arr;
242 }
243#else
244 // C++11: lazy runtime array
245 static const std::array<enum_type::string_type<Enum>, sizeof...(Is) + 1> &

247 static const std::array<enum_type::string_type<Enum>, sizeof...(Is) + 1>

250 "", enum_type::template name<Enum, Is>()...};
251

257 tmp[idx] = enum_type::string_type<Enum>(pair.second);

323template <typename Enum, int Min, int Max>
324MGUTILITY_CNSTXPR inline auto to_enum_impl(mgutility::string_view str) noexcept
325 -> mgutility::optional<Enum> {
326 MGUTILITY_CNSTXPR_CLANG_WA auto arr = get_enum_array<Enum, Min, Max>();
327
328 const auto index{detail::find(arr, str)};
329 return index == 0
330 ? mgutility::nullopt
331 : mgutility::optional<Enum>{static_cast<Enum>(index + Min - 1)};
332}

343template <typename Enum, int Min, int Max>
344MGUTILITY_CNSTXPR auto to_enum_bitmask_impl(mgutility::string_view str) noexcept
345 -> mgutility::optional<Enum> {
346
347 // Check if the string contains a '|' character
348 if (str.find('|') == mgutility::string_view::npos) {
349 return to_enum_impl<Enum, Min, Max>(str);
350 }
351
352 mgutility::optional<Enum> result{mgutility::nullopt};
353 std::size_t index = 0;
354
355 for (std::size_t i = 0; i < str.size(); ++i) {
356 if (str[i] == '|') {
357 auto name = str.substr(index, i - index);
358 auto maybe_enum = to_enum_impl<Enum, Min, Max>(name);
359
360 if (!name.empty() && maybe_enum) {
361 result.emplace(result ? static_cast<Enum>(*result | *maybe_enum)
362 : *maybe_enum);
363 }
364
365 index = i + 1;
366 }
367 }
368
369 auto maybe_enum = to_enum_impl<Enum, Min, Max>(str.substr(index));
370 if (result && maybe_enum) {
371 result.emplace(static_cast<Enum>(*result | *maybe_enum));
372 } else {
373 result.reset();
374 }
375
376 return result;
377}

388template <typename Enum, int Min, int Max,
389 detail::enable_if_t<!detail::has_bit_or<Enum>::value, bool> = true>
390MGUTILITY_CNSTXPR auto enum_name_impl(Enum enumValue) noexcept
391 -> mgutility::string_view {
392 MGUTILITY_CNSTXPR_CLANG_WA auto arr = get_enum_array<Enum, Min, Max>();
393 const auto index{(Min < 0 ? -Min : Min) + static_cast<int>(enumValue) + 1};
394 return arr[static_cast<size_t>(
395 (index < Min || index > static_cast<int>(arr.size()) - 1) ? 0 : index)];

408// NOLINTNEXTLINE [modernize-use-constraints]
409template <typename Enum, int Min, int Max,
410 detail::enable_if_t<detail::has_bit_or<Enum>::value, bool> = true>
411MGUTILITY_CNSTXPR_CLANG_WA auto enum_name_impl(Enum enumValue) noexcept
412 -> mgutility::fixed_string<enum_name_buffer<Enum>::size> {
413
414 // Get the array of enum names
415 MGUTILITY_CNSTXPR_CLANG_WA auto arr = get_enum_array<Enum, Min, Max>();
416
417 // Calculate the index in the array
418 const auto index = (Min < 0 ? -Min : Min) + static_cast<int>(enumValue) + 1;
419 const auto name =
420 arr[(index < Min || index >= static_cast<int>(arr.size())) ? 0 : index];
421
422 // Return the name if it's valid
423 if (!name.empty() && !is_digit(name[0])) {
424 return mgutility::fixed_string<enum_name_buffer<Enum>::size>{}.append(name);
425 }
426
427 // Construct bitmasked name
428 mgutility::fixed_string<enum_name_buffer<Enum>::size> bitmasked_name;
429 for (auto i = Min; i < Max; ++i) {
430 const auto idx = (Min < 0 ? -Min : Min) + i + 1;
431 if (idx >= 0 && idx < static_cast<int>(arr.size()) && !arr[idx].empty() &&
432 !detail::is_digit(arr[idx][0]) &&
433 (enumValue & static_cast<Enum>(i)) == static_cast<Enum>(i)) {
434 bitmasked_name.append(arr[idx]).append("|");
435 }
436 }
437
438 // Remove the trailing '|' if present
439 if (!bitmasked_name.empty()) {
440 bitmasked_name.pop_back();
441 }
442
443 return bitmasked_name;
444}
