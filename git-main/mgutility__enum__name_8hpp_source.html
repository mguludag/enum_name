122constexpr auto strlen_constexpr(const char *str, size_t sz = 0) noexcept
123 -> size_t {
124 // NOLINTNEXTLINE [cppcoreguidelines-pro-bounds-pointer-arithmetic]
125 return str[sz] == '\0' ? sz : strlen_constexpr(str, ++sz);
126}

134constexpr auto is_digit(char character) noexcept -> bool {
135 return character >= '0' && character <= '9';
136}

147MGUTILITY_CNSTXPR int strncmp_constexpr(const char *lhs, const char *rhs,
148 size_t count) noexcept {
149 for (size_t i = 0; i < count; ++i) {
150 // NOLINTNEXTLINE [cppcoreguidelines-pro-bounds-pointer-arithmetic]
151 if (lhs[i] != rhs[i] || lhs[i] == '\0' || rhs[i] == '\0') {
152 // NOLINTNEXTLINE [cppcoreguidelines-pro-bounds-pointer-arithmetic]
153 return static_cast<unsigned char>(lhs[i]) -
154 // NOLINTNEXTLINE [cppcoreguidelines-pro-bounds-pointer-arithmetic]
155 static_cast<unsigned char>(rhs[i]);
156 }
157 }
158 return 0;
159}

161template <typename Range, typename Pred>
162MGUTILITY_CNSTXPR auto find(const Range &rng, const Pred &pred) -> size_t {
163 for (size_t i = 1; i < rng.size(); ++i) {
164 if (pred == rng[i]) {
165 return i;
166 }
167 }
168 return 0;
169};

1283 MGUTILITY_CNSTXPR static mgutility::string_view
1284 parse(mgutility::string_view str) noexcept {
1285#if defined(__clang__) || defined(__GNUC__)
1286#if defined(__clang__)
1287 auto end = str.rfind(']');
1288#elif defined(__GNUC__) && !defined(__clang__)
1289 auto end = str.rfind(';');
1290#endif
1291 // Typical form:
1292 // "Enum = MyEnum::Value]"
1293
1294 auto pos = str.rfind('=', end);
1295 if (pos == mgutility::string_view::npos) {
1296 return {};
1297 }
1298 pos += 2; // skip "::"
1299
1300 auto result = str.substr(pos, end - pos);
1301
1302#elif defined(_MSC_VER)
1303 // MSVC: different format
1304 auto pos = str.rfind(',');
1305 if (pos == mgutility::string_view::npos)
1306 return {};
1307
1308 ++pos;
1309
1310 auto end = str.rfind('>');
1311 auto result = str.substr(pos, end - pos);
1312
1313#else
1314 return {};
1315#endif
1316
1317 // -------------------------------
1318 // 4. Validate result
1319 // -------------------------------
1320 if (result.empty()) {
1321 return {};
1322 }
1323
1324 // invalid cases look like "(Enum)123"
1325 if (result[0] == '(') {
1326 return {};
1327 }
1328
1329 return result.substr(result.rfind(':') + 1);

1440template <typename Enum, int Min, int Max>
1441MGUTILITY_CNSTXPR inline auto to_enum_impl(mgutility::string_view str) noexcept
1442 -> mgutility::optional<Enum> {
1443 MGUTILITY_CNSTXPR_CLANG_WA auto arr = get_enum_array<Enum, Min, Max>();
1444
1445 const auto index{detail::find(arr, str)};

1460template <typename Enum, int Min, int Max>
1461MGUTILITY_CNSTXPR auto to_enum_bitmask_impl(mgutility::string_view str) noexcept
1462 -> mgutility::optional<Enum> {
1463
1464 // Check if the string contains a '|' character
1465 if (str.find('|') == mgutility::string_view::npos) {
1466 return to_enum_impl<Enum, Min, Max>(str);
1467 }
1468
1469 mgutility::optional<Enum> result{mgutility::nullopt};
1470 std::size_t index = 0;
1471
1472 for (std::size_t i = 0; i < str.size(); ++i) {
1473 if (str[i] == '|') {
1474 auto name = str.substr(index, i - index);
1475 auto maybe_enum = to_enum_impl<Enum, Min, Max>(name);
1476
1477 if (!name.empty() && maybe_enum) {
1478 result.emplace(result ? static_cast<Enum>(*result | *maybe_enum)
1479 : *maybe_enum);
1480 }
1481
1482 index = i + 1;
1483 }
1484 }
1485
1486 auto maybe_enum = to_enum_impl<Enum, Min, Max>(str.substr(index));
1487 if (result && maybe_enum) {
1488 result.emplace(static_cast<Enum>(*result | *maybe_enum));
1489 } else {
1490 result.reset();
1491 }
1492
1493 return result;
1494}

1506 detail::enable_if_t<!detail::has_bit_or<Enum>::value, bool> = true>
1507MGUTILITY_CNSTXPR auto enum_name_impl(Enum enumValue) noexcept
1508 -> mgutility::string_view {

1524// NOLINTNEXTLINE [modernize-use-constraints]
1525template <typename Enum, int Min, int Max,
1526 detail::enable_if_t<detail::has_bit_or<Enum>::value, bool> = true>
1527MGUTILITY_CNSTXPR_CLANG_WA auto enum_name_impl(Enum enumValue) noexcept
1528 -> fixed_string<enum_name_buffer<Enum>::size> {
1529
1530 // Get the array of enum names
1531 MGUTILITY_CNSTXPR_CLANG_WA auto arr = get_enum_array<Enum, Min, Max>();
1532
1533 // Calculate the index in the array
1534 const auto index = (Min < 0 ? -Min : Min) + static_cast<int>(enumValue) + 1;
1535 const auto name =
1536 arr[(index < Min || index >= static_cast<int>(arr.size())) ? 0 : index];
1537
1538 // Return the name if it's valid
1539 if (!name.empty() && !is_digit(name[0])) {
1540 return fixed_string<enum_name_buffer<Enum>::size>{}.append(name);
1541 }
1542
1543 // Construct bitmasked name
1544 fixed_string<enum_name_buffer<Enum>::size> bitmasked_name;
1545 for (auto i = Min; i < Max; ++i) {
1546 const auto idx = (Min < 0 ? -Min : Min) + i + 1;
1547 if (idx >= 0 && idx < static_cast<int>(arr.size()) && !arr[idx].empty() &&
1548 !detail::is_digit(arr[idx][0]) &&
1549 (enumValue & static_cast<Enum>(i)) == static_cast<Enum>(i)) {
1550 bitmasked_name.append(arr[idx]).append("|");
1551 }
1552 }
1553
1554 // Remove the trailing '|' if present
1555 if (!bitmasked_name.empty()) {
1556 bitmasked_name.pop_back();
1557 }
1558
1559 return bitmasked_name;
1560}
